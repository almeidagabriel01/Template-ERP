**AI-05 describe block — "AI-05: Inactive module causes Lia to refuse":**

This test verifies that when a module (e.g., WhatsApp) is disabled for the tenant, Lia refuses to execute tools related to that module.

**Strategy:** The TENANT_AI_TEST has `whatsappEnabled: true`. To test an inactive module:
1. Temporarily update the tenant doc to set a specific module flag to false
2. Send a message asking Lia to use that module's tool
3. Verify Lia's response indicates the module is inactive
4. Restore the tenant doc

Since we need to pick a module to disable, check what tools exist:
- The executor MUST read `functions/src/ai/tools/index.ts` to find which tools check module flags
- WhatsApp tools (if they exist) check `whatsappEnabled` on the tenant doc

```typescript
test.describe("AI-05: Inactive module causes Lia to refuse", () => {
  test.skip(!process.env.GEMINI_API_KEY, "Requires GEMINI_API_KEY");

  const db = getTestDb();

  test.afterEach(async () => {
    // Restore whatsappEnabled to true for ai-test tenant
    await db.collection("tenants").doc("ai-test").update({
      whatsappEnabled: true,
    });
  });

  test("disabled module tool call returns refusal message", async () => {
    // Disable WhatsApp module for ai-test tenant
    await db.collection("tenants").doc("ai-test").update({
      whatsappEnabled: false,
    });

    const { idToken } = await signInWithEmailPassword(
      USER_AI_ADMIN.email,
      USER_AI_ADMIN.password,
    );

    const response = await fetch(`${FUNCTIONS_BASE}/v1/ai/chat`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${idToken}`,
      },
      body: JSON.stringify({
        message: "Enviar mensagem WhatsApp para o cliente João",
        sessionId: "test-inactive-module",
      }),
    });

    // The response should be 200 (SSE stream starts) but the AI should respond
    // with a message indicating the module is not available
    expect(response.status).toBe(200);

    const body = await response.text();
    // Parse SSE events to find the text response
    const lines = body.split("\n").filter(l => l.startsWith("data: "));
    const events = lines
      .map(l => l.replace("data: ", ""))
      .filter(l => l !== "[DONE]");

    // At least one event should contain text indicating module is unavailable
    const fullText = events.join(" ");
    
    // The AI should mention that WhatsApp is not available/inactive
    // Exact wording depends on the system prompt and tool gating
    // Check for common indicators: "desativado", "não disponível", "não habilitado", "indisponível"
    const hasRefusal = /desativad|não.*(disponível|habilit|ativ)|indisponível|módulo.*inativ/i.test(fullText);
    
    // If the tool is gated before Gemini even sees it (via buildAvailableTools),
    // Gemini won't have the tool in its schema and will reply naturally that
    // it cannot perform that action. Either way, the response should NOT contain
    // a successful tool execution result.
    expect(hasRefusal || !fullText.includes("mensagem enviada")).toBe(true);
  });
});
```

**AI-11 Group B describe block — "AI-11 Group B: Member blocked from admin tool execution":**

This test verifies that when a member user asks Lia to perform an admin-only action (e.g., delete a contact), the tool executor blocks the execution. The `buildAvailableTools()` in `functions/src/ai/tools/index.ts` filters out admin-only tools for member users (line 112: `entry.minRole === "admin" && !isAdmin`). Additionally, the executor double-checks at line 537.

Since admin-only tools are filtered from the Gemini function declarations for member users, Gemini will NOT call those tools. The test verifies this behavior end-to-end:

```typescript
test.describe("AI-11 Group B: Member blocked from admin tool execution", () => {
  test.skip(!process.env.GEMINI_API_KEY, "Requires GEMINI_API_KEY");

  const db = getTestDb();

  test.afterEach(async () => {
    // Clean up any contacts that might have been created
    const snap = await db.collection("clients")
      .where("tenantId", "==", "ai-test")
      .where("name", "==", "Contato Para Deletar")
      .get();
    for (const doc of snap.docs) {
      await doc.ref.delete();
    }
  });

  test("member user cannot execute admin-only tool (delete_contact)", async () => {
    // First, create a test contact as admin so there's something to delete
    await db.collection("clients").doc("ai-test-delete-target").set({
      tenantId: "ai-test",
      name: "Contato Para Deletar",
      email: "deletar@test.com",
      createdAt: new Date(),
      updatedAt: new Date(),
    });

    // Sign in as member
    const { idToken } = await signInWithEmailPassword(
      USER_AI_MEMBER.email,
      USER_AI_MEMBER.password,
    );

    // Ask Lia to delete the contact — delete_contact has minRole: "admin"
    // buildAvailableTools() should filter it out for member users
    const response = await fetch(`${FUNCTIONS_BASE}/v1/ai/chat`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${idToken}`,
      },
      body: JSON.stringify({
        message: "Excluir o contato Contato Para Deletar",
        sessionId: "test-member-admin-tool",
      }),
    });

    expect(response.status).toBe(200);

    const body = await response.text();
    // Parse SSE to check what happened
    const lines = body.split("\n").filter(l => l.startsWith("data: "));
    const events = lines
      .map(l => l.replace("data: ", ""))
      .filter(l => l !== "[DONE]");

    // Check that no tool_call for delete_contact was executed
    const hasDeleteToolCall = events.some(e => {
      try {
        const parsed = JSON.parse(e) as Record<string, unknown>;
        if (parsed.type === "tool_call") {
          const toolCall = parsed.toolCall as Record<string, unknown>;
          return toolCall.toolName === "delete_contact";
        }
        // Also check tool_result for the admin permission error
        if (parsed.type === "tool_result") {
          const toolResult = parsed.toolResult as Record<string, unknown>;
          return (toolResult.error as string)?.includes("permissao de administrador");
        }
        return false;
      } catch {
        return false;
      }
    });

    // If buildAvailableTools filtered out delete_contact, Gemini won't call it at all.
    // The response should be a text message saying Lia can't do that.
    // If the executor double-check caught it, there would be a tool_result with error.
    // Either way: the contact should NOT be deleted.
    const contactSnap = await db.doc("clients/ai-test-delete-target").get();
    expect(contactSnap.exists).toBe(true);

    // Cleanup
    await db.doc("clients/ai-test-delete-target").delete();
  });
});
```
