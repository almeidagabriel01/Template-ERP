```typescript
test.describe("FIN-06: Installment transactions", () => {
  test("creates installment group via API and marks installments as paid via UI", async ({ authenticatedPage }) => {
    // Step 1: Get auth token for API calls
    const { idToken } = await signInWithEmailPassword(
      USER_ADMIN_ALPHA.email,
      USER_ADMIN_ALPHA.password,
    );

    const timestamp = Date.now();

    // Step 2: Create installment transaction via backend API (per D-04)
    // CRITICAL: The executor MUST verify the exact field names by reading
    // functions/src/api/controllers/transactions.controller.ts before implementing.
    // The fields below are ASSUMED from research (assumption A1) and may need adjustment.
    const createResponse = await authenticatedPage.request.post("/api/backend/v1/transactions", {
      headers: { Authorization: `Bearer ${idToken}` },
      data: {
        type: "income",
        description: `Installment Test ${timestamp}`,
        amount: "300.00",              // total amount
        date: "2024-07-01",
        dueDate: "2024-07-01",
        wallet: "wallet-alpha-main",   // wallet ID
        isInstallment: true,
        installmentCount: 3,
        paymentMode: "total",          // total means amount is the sum, divided by installmentCount
        // Verify: does the API need additional fields?
      },
    });
    expect(createResponse.status()).toBe(201);
    const createData = await createResponse.json();
    // The API response structure needs verification — it may return:
    // { success: true, transactionId: "..." } or { id: "..." } or similar
    const transactionId = createData.transactionId || createData.id;
    expect(transactionId).toBeTruthy();

    // Step 3: Navigate to the transaction detail page to see installments
    await authenticatedPage.goto(`/transactions/${transactionId}`);
    await authenticatedPage.waitForURL(new RegExp(`/transactions/${transactionId}`), { timeout: 15000 });

    // Step 4: Verify installment rows are visible
    // TransactionInstallmentsList renders rows like "Parcela 1/3", "Parcela 2/3", "Parcela 3/3"
    await expect(authenticatedPage.getByText(/parcela 1/i)).toBeVisible({ timeout: 10000 });

    // Step 5: Mark installment 1 as paid via UI (per D-05)
    // Find the installment row's status dropdown button and click it
    // The dropdown button shows current status (e.g., "Pendente")
    // Then select "Pago" from the dropdown menu items
    // SEQUENTIAL ORDER REQUIRED (pitfall 3): must pay installment 1 before 2
    const installment1Row = authenticatedPage.locator('tr, [data-testid*="installment"]').filter({
      hasText: /parcela 1/i,
    }).first();
    // Click the status dropdown button on this row
    const statusButton1 = installment1Row.getByRole("button", { name: /pendente|pago|atrasado/i });
    await statusButton1.click();
    // Select "Pago" from dropdown
    await authenticatedPage.getByRole("menuitem", { name: /pago/i }).click();

    // Wait for status update to process
    await authenticatedPage.waitForTimeout(1000);

    // Step 6: Verify installment 1 now shows as paid
    // The status dropdown button text should change to "Pago" or similar
    // Re-locate the row and check its status
    const updatedRow1 = authenticatedPage.locator('tr, [data-testid*="installment"]').filter({
      hasText: /parcela 1/i,
    }).first();
    await expect(updatedRow1.getByText(/pago/i)).toBeVisible();

    // Step 7: Mark installment 2 as paid (after installment 1 — sequential per pitfall 3)
    const installment2Row = authenticatedPage.locator('tr, [data-testid*="installment"]').filter({
      hasText: /parcela 2/i,
    }).first();
    const statusButton2 = installment2Row.getByRole("button", { name: /pendente|pago|atrasado/i });
    await statusButton2.click();
    await authenticatedPage.getByRole("menuitem", { name: /pago/i }).click();

    await authenticatedPage.waitForTimeout(1000);

    // Verify installment 2 now shows as paid
    const updatedRow2 = authenticatedPage.locator('tr, [data-testid*="installment"]').filter({
      hasText: /parcela 2/i,
    }).first();
    await expect(updatedRow2.getByText(/pago/i)).toBeVisible();

    // Step 8: Cleanup — delete the installment group via API
    const deleteResponse = await authenticatedPage.request.delete(
      `/api/backend/v1/transactions/${transactionId}`,
      { headers: { Authorization: `Bearer ${idToken}` } },
    );
    // Accept 200 or 204 for successful deletion
    expect([200, 204]).toContain(deleteResponse.status());
  });
});
```
